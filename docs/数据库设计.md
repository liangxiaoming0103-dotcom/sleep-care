## 二、完整 SQLite DDL 建表语句

```sql
-- =====================================================
-- 1. 用户表 users
-- =====================================================
CREATE TABLE IF NOT EXISTS users (
    user_id         INTEGER PRIMARY KEY AUTOINCREMENT,  -- 用户唯一标识，系统自动生成
    phone           TEXT    NOT NULL UNIQUE,             -- 手机号码，11位数字，唯一
    password_hash   TEXT    NOT NULL,                    -- 密码哈希值（微信授权登录可为空串）
    nickname        TEXT    NOT NULL DEFAULT '用户',      -- 用户昵称，最长50字符
    avatar_url      TEXT    NULL,                        -- 头像图片URL
    gender          INTEGER NOT NULL DEFAULT 0,          -- 性别：0=未知 1=男 2=女
    birth_year      INTEGER NULL,                        -- 出生年份（4位数字）
    role            INTEGER NOT NULL DEFAULT 0,          -- 角色：0=普通用户 1=医生 2=管理员
    status          INTEGER NOT NULL DEFAULT 0,          -- 账户状态：0=正常 1=禁用
    created_at      TEXT    NOT NULL DEFAULT (datetime('now','localtime'))  -- 注册时间
);

-- =====================================================
-- 2. 设备表 devices
-- =====================================================
CREATE TABLE IF NOT EXISTS devices (
    device_id         TEXT    NOT NULL PRIMARY KEY,      -- 设备唯一序列号，真实设备为出厂编码，虚拟设备格式：VIR+16位随机字符
    user_id           INTEGER NULL,                      -- 绑定用户ID，NULL表示未绑定
    device_name       TEXT    NOT NULL DEFAULT '我的设备', -- 用户自定义设备昵称
    is_virtual        INTEGER NOT NULL DEFAULT 0,        -- 是否虚拟设备：1=虚拟 0=真实硬件
    firmware_version  TEXT    NOT NULL DEFAULT 'V1.0.0', -- 当前固件版本号
    last_active_time  TEXT    NOT NULL DEFAULT (datetime('now','localtime')),  -- 最后活跃时间
    created_at        TEXT    NOT NULL DEFAULT (datetime('now','localtime')),  -- 添加时间
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL
);

-- =====================================================
-- 3. 睡眠报告表 sleep_reports
-- =====================================================
CREATE TABLE IF NOT EXISTS sleep_reports (
    report_id         INTEGER PRIMARY KEY AUTOINCREMENT,  -- 报告唯一标识
    user_id           INTEGER NOT NULL,                   -- 所属用户ID
    device_id         TEXT    NOT NULL,                   -- 产生报告的设备ID
    report_date       TEXT    NOT NULL,                   -- 报告对应夜晚日期（如 2026-06-14）
    sleep_score       INTEGER NOT NULL DEFAULT 0,         -- 睡眠综合评分 0-100（基于时长、深睡比例、觉醒次数等）
    total_minutes     INTEGER NOT NULL DEFAULT 0,         -- 总睡眠时长（分钟）
    deep_minutes      INTEGER NOT NULL DEFAULT 0,         -- 深睡时长（分钟），正常占比15%-25%
    rem_minutes       INTEGER NOT NULL DEFAULT 0,         -- REM时长（分钟），正常占比20%-25%
    light_minutes     INTEGER NOT NULL DEFAULT 0,         -- 浅睡时长（分钟），正常占比45%-55%
    wake_minutes      INTEGER NOT NULL DEFAULT 0,         -- 夜间清醒总时长（分钟）
    avg_heart_rate    REAL    NULL,                       -- 整夜平均心率（bpm），正常范围40-100
    events_json       TEXT    NULL,                       -- 异常事件JSON数组，如 [{"type":"apnea","time":"02:30","duration":30}]
    heart_rate_curve  TEXT    NULL,                       -- 分钟级心率数据JSON数组 [75,74,72,...]
    respiration_curve TEXT    NULL,                       -- 分钟级呼吸频率JSON数组 [16,15,14,...]
    stage_curve       TEXT    NULL,                       -- 睡眠分期序列JSON数组（每30秒一个值：0清醒 1浅睡 2深睡 3REM）
    noise_curve       TEXT    NULL,                       -- 环境噪音等级曲线JSON数组（dB SPL）
    created_at        TEXT    NOT NULL DEFAULT (datetime('now','localtime')),  -- 报告生成时间
    FOREIGN KEY (user_id)   REFERENCES users(user_id)   ON DELETE CASCADE,
    FOREIGN KEY (device_id) REFERENCES devices(device_id) ON DELETE CASCADE,
    UNIQUE(user_id, report_date)                         -- 同一用户每天仅一份报告
);

-- =====================================================
-- 4. 用户设置表 user_settings
-- =====================================================
CREATE TABLE IF NOT EXISTS user_settings (
    user_id                INTEGER PRIMARY KEY,            -- 用户ID，与users表一一对应
    bedtime                TEXT    NOT NULL DEFAULT '23:00:00',   -- 期望就寝时间 HH:MM:SS
    wakeup_time            TEXT    NOT NULL DEFAULT '07:00:00',   -- 期望起床时间 HH:MM:SS
    sunrise_duration       INTEGER NOT NULL DEFAULT 10,            -- 日出程序时长（分钟），范围5-30
    sound_preference       TEXT    NOT NULL DEFAULT 'white_noise', -- 助眠音效：white_noise/wave/rain/fire
    wake_sound             TEXT    NOT NULL DEFAULT 'bird',        -- 唤醒声音：bird/music/nature
    preferred_brightness   INTEGER NOT NULL DEFAULT 50,            -- 偏好亮度 1-100（对应1-300 lux）
    preferred_volume       INTEGER NOT NULL DEFAULT 40,            -- 偏好音量 0-100（0=静音 100=85dB）
    device_timezone        TEXT    NOT NULL DEFAULT 'Asia/Shanghai',  -- 设备时区
    do_not_disturb_enabled INTEGER NOT NULL DEFAULT 0,             -- 夜间勿扰开关：0=关闭 1=开启
    dnd_start              TEXT    NOT NULL DEFAULT '23:00:00',   -- 勿扰开始时间
    dnd_end                TEXT    NOT NULL DEFAULT '06:00:00',   -- 勿扰结束时间
    created_at             TEXT    NOT NULL DEFAULT (datetime('now','localtime')),  -- 创建时间
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- =====================================================
-- 5. 医生授权表 doctor_authorizations
-- =====================================================
CREATE TABLE IF NOT EXISTS doctor_authorizations (
    id               INTEGER PRIMARY KEY AUTOINCREMENT,    -- 授权记录唯一标识
    patient_user_id  INTEGER NOT NULL,                     -- 患者用户ID（数据所有者）
    doctor_user_id   INTEGER NOT NULL,                     -- 医生用户ID（被授权者）
    status           TEXT    NOT NULL DEFAULT 'pending'    -- 状态：pending|active|expired|revoked
                        CHECK(status IN ('pending','active','expired','revoked')),
    expire_at        TEXT    NOT NULL,                     -- 授权到期时间
    created_at       TEXT    NOT NULL DEFAULT (datetime('now','localtime')),  -- 授权发起时间
    FOREIGN KEY (patient_user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (doctor_user_id)  REFERENCES users(user_id) ON DELETE CASCADE
);

-- =====================================================
-- 索引（提升查询性能）
-- =====================================================
CREATE INDEX IF NOT EXISTS idx_devices_user_id        ON devices(user_id);
CREATE INDEX IF NOT EXISTS idx_reports_user_date       ON sleep_reports(user_id, report_date);
CREATE INDEX IF NOT EXISTS idx_reports_device_id       ON sleep_reports(device_id);
CREATE INDEX IF NOT EXISTS idx_auth_patient            ON doctor_authorizations(patient_user_id);
CREATE INDEX IF NOT EXISTS idx_auth_doctor             ON doctor_authorizations(doctor_user_id);
CREATE INDEX IF NOT EXISTS idx_auth_status             ON doctor_authorizations(status);
```

## 三、字段说明（中文注释）

### 3.1 users — 用户表

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `user_id` | INTEGER | PK, AUTOINCREMENT | 用户唯一标识，系统自动生成 |
| `phone` | TEXT | NOT NULL, UNIQUE | 手机号码，用于登录和身份识别，11位数字 |
| `password_hash` | TEXT | NOT NULL | 密码哈希值，微信授权登录时可为空串 |
| `nickname` | TEXT | NOT NULL, DEFAULT '用户' | 用户昵称，可修改，最长50字符 |
| `avatar_url` | TEXT | NULL | 头像图片URL，默认使用系统头像 |
| `gender` | INTEGER | NOT NULL, DEFAULT 0 | 性别：0=未知 1=男 2=女 |
| `birth_year` | INTEGER | NULL | 出生年份（4位数字），用于年龄计算 |
| `role` | INTEGER | NOT NULL, DEFAULT 0 | 用户角色：0=普通用户 1=医生 2=管理员 |
| `status` | INTEGER | NOT NULL, DEFAULT 0 | 账户状态：0=正常 1=禁用 |
| `created_at` | TEXT | NOT NULL | 账户注册时间，默认当前时间 |

### 3.2 devices — 设备表

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `device_id` | TEXT | PK | 设备唯一序列号，硬件设备为出厂编码，虚拟设备格式：VIR+16位随机字符 |
| `user_id` | INTEGER | FK → users.user_id, NULL | 当前绑定用户的ID，NULL表示设备未绑定 |
| `device_name` | TEXT | NOT NULL, DEFAULT '我的设备' | 用户自定义设备昵称 |
| `is_virtual` | INTEGER | NOT NULL, DEFAULT 0 | 是否为虚拟设备：1=虚拟（演示/模拟） 0=真实硬件 |
| `firmware_version` | TEXT | NOT NULL, DEFAULT 'V1.0.0' | 设备当前固件版本号，格式如 V1.0.0 |
| `last_active_time` | TEXT | NOT NULL | 设备最后活跃时间（真实设备上报心跳，虚拟设备每日更新） |
| `created_at` | TEXT | NOT NULL | 设备首次添加到系统的时间 |

### 3.3 sleep_reports — 睡眠报告表

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `report_id` | INTEGER | PK, AUTOINCREMENT | 睡眠报告唯一标识 |
| `user_id` | INTEGER | FK → users.user_id, NOT NULL | 所属用户ID |
| `device_id` | TEXT | FK → devices.device_id, NOT NULL | 产生此报告的设备ID |
| `report_date` | TEXT | NOT NULL, UNIQUE(user_id, date) | 报告对应夜晚日期（如 2026-06-14） |
| `sleep_score` | INTEGER | NOT NULL, DEFAULT 0 | 睡眠综合评分 0-100，依据时长、深睡比例、觉醒次数、心率等 |
| `total_minutes` | INTEGER | NOT NULL, DEFAULT 0 | 总睡眠时长（分钟），含深睡+浅睡+REM，不含清醒时间 |
| `deep_minutes` | INTEGER | NOT NULL, DEFAULT 0 | 深睡时长（分钟），正常成人占比15%-25% |
| `rem_minutes` | INTEGER | NOT NULL, DEFAULT 0 | REM快速眼动时长（分钟），正常成人占比20%-25% |
| `light_minutes` | INTEGER | NOT NULL, DEFAULT 0 | 浅睡时长（分钟），正常成人占比45%-55% |
| `wake_minutes` | INTEGER | NOT NULL, DEFAULT 0 | 夜间总清醒时长（分钟），含入睡潜伏期和夜间觉醒总和 |
| `avg_heart_rate` | REAL | NULL | 整夜平均心率（次/分钟），正常静息心率40-100 bpm |
| `events_json` | TEXT | NULL | 异常事件JSON数组，包含type（类型）/time（时间）/duration（持续秒数） |
| `heart_rate_curve` | TEXT | NULL | 分钟级心率JSON数组，长度=总睡眠分钟数，每分钟一个整数值 |
| `respiration_curve` | TEXT | NULL | 分钟级呼吸频率JSON数组，长度=总睡眠分钟数 |
| `stage_curve` | TEXT | NULL | 睡眠分期序列JSON数组，每30秒一个值（0清醒 1浅睡 2深睡 3REM） |
| `noise_curve` | TEXT | NULL | 环境噪音等级曲线JSON数组（dB SPL），模拟数据 |
| `created_at` | TEXT | NOT NULL | 报告生成时间，通常为次日凌晨 |

### 3.4 user_settings — 用户设置表

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `user_id` | INTEGER | PK, FK → users.user_id | 用户ID，与users表一一对应 |
| `bedtime` | TEXT | NOT NULL, DEFAULT '23:00:00' | 期望就寝时间，格式 HH:MM:SS，用于模拟数据生成 |
| `wakeup_time` | TEXT | NOT NULL, DEFAULT '07:00:00' | 期望起床时间，格式 HH:MM:SS，用于模拟数据生成 |
| `sunrise_duration` | INTEGER | NOT NULL, DEFAULT 10 | 日出程序时长（分钟），范围5-30，用于晨间缓慢唤醒 |
| `sound_preference` | TEXT | NOT NULL, DEFAULT 'white_noise' | 助眠音效类型：white_noise(白噪音)/wave(海浪)/rain(雨声)/fire(篝火) |
| `wake_sound` | TEXT | NOT NULL, DEFAULT 'bird' | 唤醒声音类型：bird(鸟鸣)/music(轻柔音乐)/nature(自然音) |
| `preferred_brightness` | INTEGER | NOT NULL, DEFAULT 50 | 偏好亮度等级 1-100（对应实际亮度 1-300 lux） |
| `preferred_volume` | INTEGER | NOT NULL, DEFAULT 40 | 偏好音量等级 0-100（0=静音 100=85dB SPL） |
| `device_timezone` | TEXT | NOT NULL, DEFAULT 'Asia/Shanghai' | 设备时区，影响报告日期显示和模拟数据生成时间 |
| `do_not_disturb_enabled` | INTEGER | NOT NULL, DEFAULT 0 | 夜间勿扰开关：0=关闭 1=开启 |
| `dnd_start` | TEXT | NOT NULL, DEFAULT '23:00:00' | 勿扰模式开始时间 |
| `dnd_end` | TEXT | NOT NULL, DEFAULT '06:00:00' | 勿扰模式结束时间 |
| `created_at` | TEXT | NOT NULL | 设置首次创建时间 |

### 3.5 doctor_authorizations — 医生授权表

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| `id` | INTEGER | PK, AUTOINCREMENT | 授权记录唯一标识 |
| `patient_user_id` | INTEGER | FK → users.user_id, NOT NULL | 患者用户ID（数据所有者） |
| `doctor_user_id` | INTEGER | FK → users.user_id, NOT NULL | 医生用户ID（被授权者） |
| `status` | TEXT | NOT NULL, DEFAULT 'pending', CHECK约束 | 授权状态：pending(等待确认)/active(有效)/expired(过期)/revoked(撤销) |
| `expire_at` | TEXT | NOT NULL | 授权到期时间，默认30天后，过期后医生无法查看数据 |
| `created_at` | TEXT | NOT NULL | 授权发起时间 |

## 四、SQLite 适配说明

| MySQL 原设计 | SQLite 实现 | 说明 |
|-------------|------------|------|
| `bigint` | `INTEGER` | SQLite 整数自动64位 |
| `varchar(n)` | `TEXT` | SQLite 无 varchar，统一用 TEXT |
| `datetime` | `TEXT` | 以 ISO-8601 字符串存储 |
| `decimal(5,2)` | `REAL` | 浮点类型 |
| `enum(...)` | `TEXT` + `CHECK(...)` | SQLite 无 ENUM，用 CHECK 约束替代 |
| `boolean` | `INTEGER` (0/1) | SQLite 无布尔类型 |
| `AUTO_INCREMENT` | `AUTOINCREMENT` | 关键词不同 |
| 外键默认关闭 | `PRAGMA foreign_keys = ON;` | 需在连接时手动开启外键约束 |
